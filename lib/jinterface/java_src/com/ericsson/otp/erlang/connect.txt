// identify self and other node
OtpSelf self = new OtpSelf("me@host", "mycookie");
OtpPeer peer = new OtpPeer("him@host");

// connect to peer
OtpConnection conn = peer.connect(self);

// make a message
OtpErlangObject msg = new OtpErlangAtom("hejsan");

// send message to known server
conn.send("servername", msg);

// receive reply
OtpErlangObject p = conn.recv();

// send message to pid
if (p instanceof OtpErlangPid) {
	conn.send(p,msg);
}

// done
conn.close();
